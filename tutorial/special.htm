Special Windows

/*
 *  Window cannot be resized.
 */
#include "graphapp.h"

int main(int argc, char *argv[])
{
  App *app;
  Window *w;

  app = new_app(argc, argv);
  w = new_window(app, rect(0,0,300,250),
                 "Special Window",
                 TITLEBAR + CLOSEBOX + MINIMIZE + MAXIMIZE);
  show_window(w);
  main_loop(app);
  return 0;
}
