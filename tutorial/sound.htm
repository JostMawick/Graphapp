#include <stdio.h>
#include "graphapp.h"

App *app;

void do_it(Control *b)
{
  beep(app);
}

int main(int argc, char *argv[])
{
  Window *w;
  Control *b;

  app = new_app(argc, argv);
  w = new_window(app, rect(20,20,400,300),
                 "Listen...", STANDARD_WINDOW);
  b = new_button(w, rect(10,10,80,30), "Press", do_it);
  show_window(w);
  main_loop(app);
  return 0;
}
